# Table 1. Descriptive statistics of 6-minute walk distance before and after intervention

# Descriptive Statistics
summary_table = distance_long.groupby("time")["distance"].agg(
    Median="median",
    Q1=lambda x: x.quantile(0.25),
    Q3=lambda x: x.quantile(0.75),
    Mean="mean",
    SD="std"
)

# Rename timepoints
summary_table.index = ["T-3", "T+6"]


# Compute individual changes (delta)
improvement = (data_inclus["delta_6mwt"] > 0).sum()
no_change = (data_inclus["delta_6mwt"] == 0).sum()
decrease = (data_inclus["delta_6mwt"] < 0).sum()


# Final table with descriptive statistics and counts of improvement/no change/decrease
final_table = summary_table.copy()
# Add column for counts of improvement/no change/decrease
final_table["n"] = ""

final_table.loc["Improved"] = ["", "", "", "", "", improvement]
final_table.loc["No change"] = ["", "", "", "", "", no_change]
final_table.loc["Decreased"] = ["", "", "", "", "", decrease]

# Export the final table to PDF
import matplotlib.pyplot as plt

fig, ax = plt.subplots()
ax.axis('off')

table = ax.table(
    cellText=final_table.round(1).values,
    colLabels=final_table.columns,
    rowLabels=final_table.index,
    loc='center'
)

# Color empty cells in grey
for i in range(len(final_table)):
    for j in range(len(final_table.columns)):
        if final_table.iloc[i, j] == "" or pd.isna(final_table.iloc[i, j]):
            table[(i+1, j)].set_facecolor("#D3D3D3")

# Add title
plt.title(
    "Table 1. Distribution of 6-minute walk distance before and after intervention",
    fontsize=12,
    pad=20
)

plt.savefig("figures/ Summary of 6-minute walk distance at baseline and post-intervention.pdf", bbox_inches="tight")
plt.close()
